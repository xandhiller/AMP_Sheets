\eeq 

\begeq{ratio_test}
  \text{Ratio test: \ \ \ }
  \lim_{n \rightarrow \infty} \bigg| \frac{z_{n+1}}{z_n}  \bigg| = L
\eeq

\begeq{root_test}
  \text{Root test: \ \ \ }
  \lim_{n \rightarrow \infty} \sqrt[n]{|z^n|} \ = L
\eeq \\
